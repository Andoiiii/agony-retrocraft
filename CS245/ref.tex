  \section*{Essential Laws of Propositional Logic}
  \begin{align*}
    \law{Double Negation}{\lnot(\lnot p) \tequiv p}
    \law{Excluded Middle}{p \lor \lnot p \tequiv 1}
    \law{Contradiction}{p \land \lnot p \tequiv 0}
    \law{Idempotence}{
    p \land p                          & \tequiv p                            \\
    p \lor p                           & \tequiv p                            \\
    }
    \law{Identity}{
    p \land 1                          & \tequiv p                            \\
    p \lor 0                           & \tequiv p                            \\
    }
    \law{Domination}{
    p \land 0                          & \tequiv 0                            \\
    p \lor 1                           & \tequiv 1                            \\
    }
    \law{Commutativity}{
    p \land q                          & \tequiv q \land p                    \\
    p \lor q                           & \tequiv q \lor p                     \\
    p \harr q                          & \tequiv q \harr p                    \\
    }
    \law{Associativity}{
    p \land (q \land r)                & \tequiv (p \land q) \land r          \\
    p \lor (q \lor r)                  & \tequiv (p \lor q) \lor r            \\
    }
    \law{Distributivity}{
    p \land (q \lor r)                 & \tequiv (p \land q) \lor (q \land r) \\
    p \lor (q \land r)                 & \tequiv (p \lor q) \land (q \lor r)  \\
    }
    \law{Implication}{p \to q \tequiv \lnot p \lor q}
    \law{Contrapositive}{p \to q \tequiv \lnot q \to \lnot p}
    \law{Equivalence}{p \harr q \tequiv (p \to q) \land (q \to p)}
    \law{De Morgan}{
    \lnot(p \land q)                   & \tequiv \lnot p \lor \lnot q         \\
    \lnot(p \lor q)                    & \tequiv \lnot p \land \lnot q        \\
    }
    \law{Absorption I}{
    p \land (p \lor q)                 & \tequiv p                            \\
    p \lor (p \land q)                 & \tequiv p                            \\
    }
    \law{Absorption II}{
    (p \lor q) \land (\lnot p \lor q)  & \tequiv q                            \\
    (p \land q) \lor (\lnot p \land q) & \tequiv q                            \\
    }
  \end{align*}

  \section*{Definitions}

  \dfn{Propositional language}{02}{3}
  A set of all strings of proposition, connective, and punctuation symbols.
  $\Lp$ has $p,q,r,\ldots,\lnot,\land,\lor,\to,\harr,(,)$.

  \dfn{Expression}{02}{4} An element of $\Lp$.
  The empty expression is $\epsilon$.

  \dfn{Segment}{02}{5} $V$ segment of $U$ if $U = W_1VW_2$.
  Initial if $W_1 = \epsilon$, final if $W_2 = \epsilon$.
  Proper if $V \neq U$.

  \dfn{Atom}{02}{6}
  $\Atom(\Lp)$ has expressions that are one proposition symbol.

  \dfn{Formula}{02}{6}
  $\Form(\Lp)$ defined by: $\Atom(\Lp) \subset \Form(\Lp)$. \\
  $A,B\in\Form(\Lp) \Rarr (\lnot A), (A \land B), (A \lor B), (A \to B), (A \harr B) \in \Form(\Lp)$. \\
  No other expressions are in $\Form(\Lp)$.

  \dfn{Scope}{02}{45}
  In $(\lnot A)$, $A$ is the scope of $\lnot$.
  In $(A \star B)$, $A$ is the left scope and $B$ is the right scope of $\star$.

  \dfn{Truth valuation}{03}{6} A function $t : \Atom(\Lp)\to\{0,1\}$.

  \dfn{Satisfiablity}{03}{9}
  The truth valuation $t$ satisfies $A$ if $A^t = 1$.
  $\S$ is satisfiable if there exists $t$ such that $\S^t = 1$.

  \dfn{Tautology}{03}{14} For all $t$, $A^t = 1$. \\
  \dfn{Contradiction}{03}{14} For all $t$, $A^t = 0$. \\
  \dfn{Contingent}{03}{14} $A$ is neither a tautology nor contradiction.

  \dfn{Tautological consequence}{03}{21}
  $\S \timplies A$ if for all $t$, $\S^t = 1$ gives $A^t = 1$. \\
  \dfn{Tautological equivalence}{03}{25}
  $A \tequiv B$ if $A \timplies B$ and $B \timplies A$.

  \dfn{Literal}{04}{10}
  A formula of the form $p$ or $\lnot p$.

  \dfn{Disjunctive clause}{04}{12}
  Disjunction with literal disjuncts. \\
  \dfn{Conjunctive clause}{04}{12}
  Conjunction with literal conjuncts.

  \dfn{DNF}{04}{13} Disjunction with conjunctive clause disjuncts. \\
  \dfn{CNF}{04}{13} Conjunction with disjunctive clause conjuncts.

  \dfn{Definability/Reducibility}{05}{2}
  Connective $\star$ reducibile to set $\mathcal S$ if $A \star B \tequiv C$
  where $C$ uses only $A$, $B$, and connectives in $\mathcal S$.

  \dfn{Adequate}{05}{7}
  Connectives which express any truth table/connective.

  \dfn{Formal deducibility}{06}{15}
  $\S \t A$ generated by finite deduction rules. \\
  \dfn{Syntactic equivalence}{06}{15}
  $A \tt B$ if $A \t B$ and $B \t A$.

  \dfn{Consistency}{06}{67}
  There is no $F$ such that $\S\t F$ and $\S\t\lnot F$.

  \section*{Theorems (...the general kind)}

  \lem{02}{29} Every formula has equal number of left/right parentheses.

  \thm[Unique Readability Theorem]{02}{32}
  Every formula is exactly one form of exactly one of
  $(\lnot A)$, $(A \land B)$, $(A \lor B)$, $(A \to B)$, $(A \harr B)$.

  \lem{03}{23} Equivalent statements of $\{A_i\} \timplies C$: \\
  Argument with premises $A_i$ and conclusion $C$ is valid.\\
  $(\bigwedge A_i) \to C$ is a tautology;
  $(\lnot C \land \bigwedge A_i)$ is a contradiction. \\
  Formula $(\lnot C \land \bigwedge A_i)$ or set $\{\lnot C, A_i\}$ is not satisfiable.

  \thm[Replaceability of tautologically equivalent formulas]{03}{44}
  If $A \tequiv A'$ and $A$ is a subformula of $B$,
  then $B \tequiv B'$ where $B'$ is $B$ with some of the $A$s replaced by $A'$.

  \thm[Duality Theorem]{03}{44}
  If $A$ has only $\lnot$, $\land$, $\lor$ and
  $\Delta(A)$ replaces atoms with negations and swaps $\land$ with $\lor$,
  then $\lnot A \tequiv \Delta(A)$.

  \thm{04}{22} All formulas have $F \tequiv \operatorname{DNF}(F)$ based on truth table. \\
  \thm{04}{24} All formulas have $F \tequiv \operatorname{CNF}(F) = \Delta(\operatorname{DNF}(\lnot F))$.

  \thm{05}{8} The set $S_0 = \{\lnot, \land, \lor\}$ is adequate.

  \thm[Finiteness of Premise Set]{06}{31}
  If $\S \t A$, $\S^0 \t A$ with finite $\S^0 \subseteq \S$.

  \thm[Soundness Theorem]{06}{45} If $\S \t A$ then $\S \timplies A$. \\
  \thm[Completeness Theorem]{06}{49} If $\S \timplies A$ then $\S \t A$.

  \lem{06}{67} $\S$ is satisfiable if and only if $\S$ is consistent.
